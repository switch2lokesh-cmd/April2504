import{expect, test}from"@playwright/test";
test("lokesh",async({page})=>
{
await page.goto("https://letcode.in/forms",{timeout:30000});
await expect(page).toHaveTitle("Forms | LetCode with Koushik");
await expect(page).toHaveURL("https://letcode.in/forms");
const header=await page.getByText("Form",{exact:true}).innerText();
console.log(header);
expect(header).toContain("Form");
await page.locator('#firstname').fill('Lokesh');
});
